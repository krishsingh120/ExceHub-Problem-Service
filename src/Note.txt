3. Testing the api using scripts
 > unit testing

   > Basic syntax for tests

      // Test that the status code is 200.
            pm.test("Status code is 200", function() {
            pm.response.to.have.status(200);
      })

      // Test that the response is in JSON format.
            pm.test("Content-type is application/json", function() {
            pm.expect(pm.response.headers.get('Content-Type')).to.include('application/json');
      });

      pm.test("Response body is an array", function() {
            let jsonData = pm.response.json();
            pm.expect(jsonData).to.be.an('array');
      })

      pm.test("Response structure is valid", function() {
            let jsonData = pm.response.json();

            pm.expect(jsonData).to.be.an('object');
            pm.expect(jsonData).to.have.all.keys('success', 'message', 'error', 'data' );

            jsonData.data.forEach((problem) => {
                  pm.expect(problem).to.have.all.keys('_id', 'title', 'description', 'difficulty', 'testCases', 'editorial', 'createdAt', 'updatedAt', '__v');
            })
      })
